/*

Type casting 

1. implicite
2. Explicite

 implicite -->     byte -> short -> int -> long -> float -> double

Explicite -->    byte <- short <- int <- long <- float <- double

*/

class TypeCasting{

	public static void main(String args[])
	{
			byte b=45;
			float f = b;
			System.out.println(f);

	}

}
